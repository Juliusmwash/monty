#include "monty.h"
#define MAX_LINE_LENGTH 100

int element;
int main(int argc, char *argv[])
{
	size_t pid, check;
	FILE *textfile;
	char line[MAX_LINE_LENGTH], *result;

	if (argc < 2)
	{
		fprintf(stderr, "USAGE: monty file\n");
		exit(EXIT_FAILURE);
	}
	textfile = fopen(argv[1], "r");
	if (textfile == NULL)
	{
		fprintf(stderr, "Error: Can't open file %s\n", argv[1]);
		exit(EXIT_FAILURE);
	}
	while(result = fgets(line, MAX_LINE_LENGTH, textfile), result != NULL)
	{
		printf("%s", line);
	}
	if(!result)
	{
	    //it may be read error or end of file
	    if(feof(textfile))
	    {
		    //it is end of file, normal exit from for
		    printf("It is end of file\n");
	    }
	    else
	    {
		    //it is i/o error
		    printf("Error not end of file\n");
	    }
    }
	return (0);
}
