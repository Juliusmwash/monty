#include <stdio.h>

#define MAX_LINE_LENGTH 1000

int main() {
    FILE *textfile;
    char line[MAX_LINE_LENGTH], *result;
    
    textfile = fopen("readme.txt", "r");
    if(textfile == NULL)
    {
	    printf("File unavailable\n");
        return 1;
    }
    
    result = fgets(line, MAX_LINE_LENGTH, textfile);
    while(result)
    {
        printf("%s", line);
    }
  if(!result)
    {//it may be read error or end of file
      if(feof(result))
        {//it is end of file, normal exit from for
         printf("It is end of file\n");
        }
       else
        {//it is i/o error
         printf("Error not end of file\n");
        }
    }
    
    fclose(textfile);
    return 0;
}
